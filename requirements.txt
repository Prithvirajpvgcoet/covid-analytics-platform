@app.get("/api/describe")
def describe_dataset():
    desc = df.describe().fillna(0).to_dict()
    # basic info usually printed, let's capture it
    buf = io.StringIO()
    df.info(buf=buf)
    info_str = buf.getvalue()
    
    info = {
        "columns": list(df.columns),
        "shape": df.shape,
        "describe": desc,
        "info": info_str
    }
    return info

@app.get("/api/statewise-first-dose")
def statewise_first_dose():
    res = state_df.groupby('State')['First Dose Administered'].max().fillna(0).to_dict()

    res = dict(sorted(res.items(), key=lambda item: item[1], reverse=True))

@app.get("/api/gender-vaccination")
def gender_vaccination():
    males = float(state_df.groupby('State')['Male(Individuals Vaccinated)'].max().sum())
    females = float(state_df.groupby('State')['Female(Individuals Vaccinated)'].max().sum())
    
    if pd.isna(males) or pd.isna(females) or (males == 0 and females == 0):
        # Fallback to older column names if needed
        males = float(state_df.groupby('State')['Male (Doses Administered)'].max().sum())
        females = float(state_df.groupby('State')['Female (Doses Administered)'].max().sum())
        
    # Replace nan with 0
    males = males if not pd.isna(males) else 0
    females = females if not pd.isna(females) else 0
    
    return {"males": males, "females": females}

@app.get("/api/ml/predict-trend")
def predict_trend():
    clean_india = india_df.dropna(subset=['Total Individuals Vaccinated']).copy()
    if clean_india.empty:
        return {"error": "Not enough data"}
    
    clean_india['Updated On'] = pd.to_datetime(clean_india['Updated On'], format='%d/%m/%Y')
    clean_india = clean_india.sort_values('Updated On')
    
    start_date = clean_india['Updated On'].min()
    clean_india['days'] = (clean_india['Updated On'] - start_date).dt.days
    
    X = clean_india[['days']].values
    y = clean_india['Total Individuals Vaccinated'].values
    
    # Time series train/test split (80% train, 20% test)
    split_idx = int(len(X) * 0.8)
    if split_idx == 0:
        split_idx = len(X) - 1
        
    X_train, X_test = X[:split_idx], X[split_idx:]
    y_train, y_test = y[:split_idx], y[split_idx:]
    
    models_configs = [
        {"name": "Linear Reg", "model": LinearRegression()},
        {"name": "Ridge (a=1)", "model": Ridge(alpha=1.0)},
        {"name": "Lasso (a=1)", "model": Lasso(alpha=1.0)},
        {"name": "Poly (Deg 2)", "model": make_pipeline(PolynomialFeatures(2), LinearRegression())},
        {"name": "Decision Tree Base", "model": DecisionTreeRegressor(max_depth=5)},
        {"name": "Random Forest", "model": RandomForestRegressor(n_estimators=15, random_state=42)}
    ]
    
    results = []
    for m_info in models_configs:
        model = m_info["model"]
        model.fit(X_train, y_train)
        
        train_preds = model.predict(X_train)
        train_r2 = r2_score(y_train, train_preds)
        
        # In case test data is empty or too small
        if len(X_test) > 0:
            test_preds = model.predict(X_test)
            test_r2 = r2_score(y_test, test_preds)
        else:
            test_r2 = 0
            
        m_info["model_full"] = m_info["model"]
        m_info["model_full"].fit(X, y) # Retrain on full for future
        
        results.append({
            "name": m_info["name"],
            "train_r2": train_r2,
            "test_r2": test_r2,
            "model_full": m_info["model_full"]
        })
        
    results.sort(key=lambda x: x["test_r2"], reverse=True) # Higher is better
    top_2 = results[:2]
    best_model_name = top_2[0]["name"]
    second_best_model_name = top_2[1]["name"]
    
    last_day = clean_india['days'].max()
    future_days = np.array([[last_day + i] for i in range(1, 31)])
    
    # 1. Predictions for the Overview Tab (using best model)
    best_model = top_2[0]["model_full"]
    future_predictions = best_model.predict(future_days)
    
    dates = [str(start_date + pd.Timedelta(days=int(d[0]))) for d in future_days]
    res = [{"date": d[:10], "prediction": max(0, p)} for d, p in zip(dates, future_predictions)]
    historical = [{"date": str(d)[:10], "actual": a} for d, a in zip(clean_india['Updated On'], y)]
    
    # 2. Predictions for top 2 models for the New Webpage
    full_days = np.vstack([X, future_days])
    full_dates = [str(start_date + pd.Timedelta(days=int(d[0])))[:10] for d in full_days]
    
    model1_preds = top_2[0]["model_full"].predict(full_days)
    model2_preds = top_2[1]["model_full"].predict(full_days)
    
    top2_chart_data = []
    for i, date_str in enumerate(full_dates):
        item = {"date": date_str}
        if i < len(y):
            item["actual"] = y[i]
        item[best_model_name] = max(0, model1_preds[i])
        item[second_best_model_name] = max(0, model2_preds[i])
        top2_chart_data.append(item)
    
    models_summary = [{"name": r["name"], "train_r2": r["train_r2"], "test_r2": r["test_r2"]} for r in results]
    
    return {
        "best_model": best_model_name,
        "second_best_model": second_best_model_name,
        "historical": historical,
        "predictions": res,
        "top2_chart_data": top2_chart_data,
        "models_summary": models_summary
    }
